#!/bin/bash

a=20
b=10

echo `expr $a + $b`
echo `expr $a - $b`
echo `expr $a \* $b`
echo `expr $a / $b`
echo `expr $a % $b`

c=20
d=10

if [ $c -eq $d -o $c -ge $d ]
then
echo "$c == $d or $c >= $d"
fi

if [ $c -gt $d ]
then
echo "$c > $d"
fi

if [ $c -ge $d ]
then
echo "$c >= $d"
fi

if [ $c -le $d -a $c -ne $d ]
then
echo "$c <= $d and $c != $d"
fi

if [ $c -ne $d ]
then
echo "$c != $d"
fi

e='this'
f='this'
if [ $e = $f ]
then
echo "$e == $f"
fi

if [ $e != $f ]
then
echo "$e != $f"
fi

if [ -z $e ]
then
echo "$e is 0"
fi

if [ -n $e ]
then
echo "$e is not 0"
fi

if [ $e ]
then
echo "$e is not null"
fi
